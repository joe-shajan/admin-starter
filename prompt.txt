generator client {
  provider = "prisma-client-js"
}

datasource db {
  provider = "mongodb"
  url      = env("DATABASE_URL") // uses connection pooling
}

enum Role {
  SUPERADMIN
  ADMIN
}

model User {
  id       String    @id @default(auto()) @map("_id") @db.ObjectId
  email    String    @unique
  password String?
  role     Role      @default(ADMIN)
  sections Section[]
}

enum ContentType {
  IMAGE
  VIDEO
  EMBEDDED
}

enum SectionType {
  SINGLE
  MULTIPLE
}

model SectionItem {
  id          String      @id @default(auto()) @map("_id") @db.ObjectId
  heading1    String
  heading2    String?
  text1       String
  text2       String?
  contentType ContentType
  url         String?
  Section     Section?    @relation(fields: [sectionsId], references: [id])
  sectionsId  String?     @db.ObjectId
}

model Section {
  id          String        @id @default(auto()) @map("_id") @db.ObjectId
  name        String        @unique
  sectionType SectionType
  sectionItem SectionItem[]
  position    Int?
  addedTime   DateTime      @default(now())
  updatedTime DateTime?
  User        User          @relation(fields: [userId], references: [id])
  userId      String        @db.ObjectId
}
